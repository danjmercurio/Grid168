def convert2Decimal(number)
        num, dec = number.split(".")
        if dec.length < 2
                dec += "0"
        end
        num + "." + dec
end

def get_type(index)
        result = ""
        outlet_types.each do |k, v|
                if v == index
                        result = k
                        break
                end
        end

        result
end
